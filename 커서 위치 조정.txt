#include <stdio.h>
#include <conio.h>
#include <Windows.h>

void gotoxy(int x, int y)
{
	COORD Pos;
	Pos.X = x;
	Pos.Y = y;
	SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), Pos);
}

{

	int x = 9;
	int y = 2;

	gotoxy(x, y);
	printf("¦£");

	for (i = 1; i < 22; i++)

		y++;
		gotoxy(x, y);
		printf("¦¢");

		if (i == 12)
		{
			gotoxy(35, i);
			printf("~Hello~");
		}

}
